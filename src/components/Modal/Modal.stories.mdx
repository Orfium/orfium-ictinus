<Modal
    open={true}
    onClose={closeFunc}
    fixedContent={true}
    contentProps={
        label: 'Label',
        heading: 'Heading',
        message:
            'Lorem ipsum dolor sit amet, consectetur adipiscing elit. Maecenas tristique eros at fringilla fringilla. Donec volutpat lobortis euismod',
        primaryCTALabel: 'Do Action',
        primaryCTA: actionCallback,
        secondaryCTALabel: 'Cancel',
        secondaryCTA: cancelCallback,
