## Toast Notification

### Toast Props

<Props of={Toast} />

### NotificationVisual Props

<Props of={NotificationVisual} />

### Usage

```js
import { Toast, NotificationVisual } from '@orfium/ictinus';

<Toast message="message" type="info" expanded closeCTA={closeCTA}>
  <NotificationVisual
    title="title"
    primaryCTALabel="Primary CTA Label"
    primaryCTA={primaryCTA}
    secondaryCTALabel="Secondary CTA Label"
    secondaryCTA={secondaryCTA}
    description="Lorem ipsum dolor sit amet, consectetur adipiscing elit. Sed viverra neque nec est porttitor"
  />
</Toast>;
```

<Preview>
  <Story name="Toast Notification" parameters={{ decorators: [withKnobs] }}>
    <PresentComponent name="" width={768}>
      <Toast
        message={text('message', 'Informative Message')}
        type={select('type', ['success', 'error', 'info', 'warning'], 'success')}
        expanded
        closeCTA={() => console.log('close action clicked')}
      >
        <NotificationVisual
          title={text('title', 'Message heading')}
          primaryCTALabel={text('primaryCTALabel', 'Primary CTA')}
          primaryCTA={() => console.log('primary action clicked')}
          secondaryCTALabel={text('secondaryCTALabel', 'Secondary CTA')}
          secondaryCTA={() => console.log('secondary action clicked')}
          description={text(
            'description',
            'Lorem ipsum dolor sit amet, consectetur adipiscing elit. Sed viverra neque nec est porttitor'
          )}

      </Toast>

  </Story>
</Preview>
