## Toast Notification

### Toast Props

<Props of={Toast} />

### NotificationVisual Props

<Props of={NotificationVisual} />

### Usage

```js
import { Toast, NotificationVisual } from '@orfium/ictinus';

<NotificationsContainer position="bottom-right">
  <Toast message="message" type="info" expanded hasMinimumHeight={false} closeCTA={closeCTA}>
    <NotificationVisual
      title="title"
      primaryCTALabel="Primary CTA Label"
      primaryCTA={primaryCTA}
      secondaryCTALabel="Secondary CTA Label"
      secondaryCTA={secondaryCTA}
      description="Lorem ipsum dolor sit amet, consectetur adipiscing elit. Sed viverra neque nec est porttitor"
    />
  </Toast>
</NotificationsContainer>;
```
