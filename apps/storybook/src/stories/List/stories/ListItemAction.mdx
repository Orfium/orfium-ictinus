## Overview

The ListItemAction component represents an action or a set of actions associated with a ListItem. It can contain buttons, icons, or any other interactive elements.

## Usage

<UsageGuidelines guidelines={['use only for buttons, icons, or any other interactive elements']} />

```
import { List, ListItem, ListItemAction, Radio, Checkbox } from '@orfium/ictinus';

const MyComponent = () => {
  return (
    <List>
      <ListItem>
        <ListItemAction>
          <RadioGroup>
            <Radio value="test1" />
          </RadioGroup>
        </ListItemAction>
      </ListItem>
      <ListItem>
        <ListItemAction>
          <Checkbox isSelected={false} isDisabled={false} />
        </ListItemAction>
      </ListItem>
    </List>
  );
};
```
